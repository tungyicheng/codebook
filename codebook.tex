// Topological Sort - 基礎版本
void solve(){
int n, m; cin >> n >> m;
vector<int> edge[m + 5];
vector<int> deg(n + 5, 0);
vector<int> ans;
for(int i = 1; i <= m; i++) {
  int u, v; cin >> u >> v;
  edge[u].push_back(v);
  deg[v]++;
}
queue<int> que;
for(int i = 1; i <= n; i++)
  if(deg[i] == 0) que.push(i);
while (!que.empty()) {
  int u = que.front(); que.pop();
  ans.push_back(u);
  for (int i : edge[u]) {
    deg[i]--;
    if (!deg[i]) que.push(i);
  }
}
if(ans.size() != n) cout << "IMPOSSIBLE";
else for(int i = 0; i < n; i++) cout << ans[i] << " \n"[i == n];
}
// Topological Sort - DP版本 (計數路徑)
void solve_dp(){
int n, m; cin >> n >> m;
vector<int> edge[n + 5];
vector<int> deg(n + 5, 0);
vector<long long> dp(n + 5, 0);
for(int i = 0; i < m; i++) {
  int u, v; cin >> u >> v;
  edge[u].push_back(v);
  deg[v]++;
}
dp[1] = 1;
queue<int> q;
for (int i = 1; i <= n; i++)
  if (deg[i] == 0) q.push(i);
while (!q.empty()) {
  int u = q.front(); q.pop();
  for (int i : edge[u]) {
    dp[u] %= (int)1e9 + 7;
    dp[i] = dp[i] + dp[u];
    dp[i] %= (int)1e9 + 7;
    deg[i]--;
    if (deg[i] == 0) q.push(i);
  }
}
dp[n] %= (int)1e9 + 7;
cout << dp[n];
}
